    # compare 方法

    接口 function compare(a: any, b: any): Boolean;

    版本 1.0.0

    # 作用：判断两个变量是否相等, 此方法用于相同数据类型的变量比较

    // 引入
import { compare } from 'project-libs';

// 单独引入
import compare from 'project-libs/build/data/compare';

    # 参数

    参数 类型 是否必须 含义
    a any 是 元素
    b any 是 元素

    # 返回值

    是否有返回值 类型 说明
    是 boolean 相同为true，不同为false

    # 小例子

    // 使用
var a = 10;
var b = 20;

compare(a, b)

// false
